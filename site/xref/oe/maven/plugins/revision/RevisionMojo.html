112     public void execute() throws MojoExecutionException, MojoFailureException {
113         if ( entries == null ) {
114             logDebug( "entries configuration is not specified, creating default entry" );
115             // defaulting to the path = project.basedir, prefix = project.artifactId, recursive, reporting unversioned
116             entries = new Entry[] {
117                     new Entry( project.getBasedir(), project.getArtifactId() ),
118             };
119         } else if ( entries.length == 0 ) {
120             // does not happen with Maven 2.2.1
121             // happens with Maven 3.0 alphas when <entries> is present but has no <entry>'s in it
122             // the configuration-entriesWithoutEntry it will fail on Maven 3.0 
123             throw new MojoExecutionException( "entries list is empty" );
124         }
125 
126         SVNStatusClient statusClient = SVNClientManager.newInstance().getStatusClient();
127 
128         for ( Entry entry : entries ) {
129             if ( entry.getPath() == null ) {
130                 logDebug( "entry path is not specified, using project.basedir: " + project.getBasedir() );
131                 entry.setPath( project.getBasedir() );
132             }
133             if ( entry.getPrefix() == null ) {
134                 logDebug( "entry properties prefix is not specified, using project.artifactId: " + project.getArtifactId() );
135                 entry.setPrefix( project.getArtifactId() );
136             }
137             entry.validate();
138             Map<String, Object> entryProperties = getEntryProperties( entry, statusClient );
139             setProjectProperties( entry.getPrefix(), entryProperties );
140         }
141     }

143     private Map<String, Object> getEntryProperties( Entry entry, SVNStatusClient statusClient ) throws MojoExecutionException {
144         logInfo( "inspecting " + entry.getPath() );
145         logDebugInfo( "  properties prefix = " + entry.getPrefix() );
146         logDebugInfo( "  recursive = " + entry.isRecursive() );
147         logDebugInfo( "  report unversioned = " + entry.reportUnversioned() );
148         logDebugInfo( "  report ignored = " + entry.reportIgnored() );
149         logDebugInfo( "  report out-of-date = " + entry.reportOutOfDate() );
150 
151         SVNStatus svnStatus;
152         try {
153             svnStatus = statusClient.doStatus( entry.getPath(), false );
154         } catch ( SVNException ignored ) {
155             // the entry path is not under version control
156             svnStatus = null;
157         }
158 
159         Map<String, Object> properties = new LinkedHashMap<String, Object>();
160         if ( svnStatus == null ) {
161             logDebugInfo( " the path is not under version control" );
162 
163             properties.put( "repository", "" );
164             properties.put( "path", "" );
165             properties.put( "revision", -1L );
166             properties.put( "mixedRevisions", "false" );
167             properties.put( "committedRevision", -1L );
168             properties.put( "status", EntryStatusSymbols.DEFAULT.getStatusSymbol( SVNStatusType.STATUS_UNVERSIONED ) );
169             properties.put( "specialStatus", EntryStatusSymbols.SPECIAL.getStatusSymbol( SVNStatusType.STATUS_UNVERSIONED ) );
170         } else {
171             SVNEntry svnEntry = svnStatus.getEntry();
172             String repositoryRoot = svnEntry.getRepositoryRoot();
173             String repositoryPath = svnEntry.getURL().substring( repositoryRoot.length() );
174             if ( repositoryPath.startsWith( "/" ) ) {
175                 repositoryPath = repositoryPath.substring( 1 );
176             }
177 
178             EntryStatusHandler entryStatusHandler = new EntryStatusHandler();
179             try {
180                 logDebugInfo( " collecting status information" );
181                 statusClient.doStatus( entry.getPath(),
182                         SVNRevision.UNDEFINED, entry.isRecursive() ? SVNDepth.INFINITY : SVNDepth.EMPTY,
183                         entry.reportOutOfDate(), true, entry.reportIgnored(), false,
184                         entryStatusHandler,
185                         null );
186             } catch ( SVNException e ) {
187                 throw new MojoExecutionException( e.getMessage(), e );
188             }
189 
190             properties.put( "repository", repositoryRoot );
191             properties.put( "path", repositoryPath );
192             properties.put( "revision", entryStatusHandler.getMaximumRevisionNumber() );
193             properties.put( "mixedRevisions", entryStatusHandler.isMixedRevisions() );
194             properties.put( "committedRevision", entryStatusHandler.getMaximumCommittedRevisionNumber() );
195             properties.put( "status", constructStatus( entry, entryStatusHandler.getLocalStatusTypes(), entryStatusHandler.getRemoteStatusTypes(), EntryStatusSymbols.DEFAULT ) );
196             properties.put( "specialStatus", constructStatus( entry, entryStatusHandler.getLocalStatusTypes(), entryStatusHandler.getRemoteStatusTypes(), EntryStatusSymbols.SPECIAL ) );
197         }
198         return properties;
199     }

201     private String constructStatus( Entry entry, Set<SVNStatusType> localStatusTypes, Set<SVNStatusType> remoteStatusTypes, EntryStatusSymbols symbols ) {
202         StringBuilder status = new StringBuilder();
203 
204         localStatusTypes.remove( SVNStatusType.STATUS_NONE );
205         localStatusTypes.remove( SVNStatusType.STATUS_NORMAL );
206         if ( localStatusTypes.remove( SVNStatusType.STATUS_ADDED ) ) {
207             status.append( symbols.getStatusSymbol( SVNStatusType.STATUS_ADDED ) );
208         }
209         if ( localStatusTypes.remove( SVNStatusType.STATUS_CONFLICTED ) ) {
210             status.append( symbols.getStatusSymbol( SVNStatusType.STATUS_CONFLICTED ) );
211         }
212         if ( localStatusTypes.remove( SVNStatusType.STATUS_DELETED ) ) {
213             status.append( symbols.getStatusSymbol( SVNStatusType.STATUS_DELETED ) );
214         }
215         if ( localStatusTypes.remove( SVNStatusType.STATUS_IGNORED ) && entry.reportIgnored() ) {
216             status.append( symbols.getStatusSymbol( SVNStatusType.STATUS_IGNORED ) );
217         }
218         if ( localStatusTypes.remove( SVNStatusType.STATUS_MODIFIED ) ) {
219             status.append( symbols.getStatusSymbol( SVNStatusType.STATUS_MODIFIED ) );
220         }
221         if ( localStatusTypes.remove( SVNStatusType.STATUS_REPLACED ) ) {
222             status.append( symbols.getStatusSymbol( SVNStatusType.STATUS_REPLACED ) );
223         }
224         if ( localStatusTypes.remove( SVNStatusType.STATUS_EXTERNAL ) ) {
225             status.append( symbols.getStatusSymbol( SVNStatusType.STATUS_EXTERNAL ) );
226         }
227         if ( localStatusTypes.remove( SVNStatusType.STATUS_UNVERSIONED ) && entry.reportUnversioned() ) {
228             status.append( symbols.getStatusSymbol( SVNStatusType.STATUS_UNVERSIONED ) );
229         }
230         if ( localStatusTypes.remove( SVNStatusType.STATUS_MISSING ) ) {
231             status.append( symbols.getStatusSymbol( SVNStatusType.STATUS_MISSING ) );
232         }
233         if ( localStatusTypes.remove( SVNStatusType.STATUS_INCOMPLETE ) ) {
234             status.append( symbols.getStatusSymbol( SVNStatusType.STATUS_INCOMPLETE ) );
235         }
236         if ( localStatusTypes.remove( SVNStatusType.STATUS_OBSTRUCTED ) ) {
237             status.append( symbols.getStatusSymbol( SVNStatusType.STATUS_OBSTRUCTED ) );
238         }
239         if ( !localStatusTypes.isEmpty() ) {
240             // future proofing
241             logWarning( "the following svn statuses are not taken into account: " + localStatusTypes );
242         }
243 
244         remoteStatusTypes.remove( SVNStatusType.STATUS_NONE );
245         if ( !remoteStatusTypes.isEmpty() && entry.reportOutOfDate() ) {
246             status.append( symbols.getOutOfDateSymbol() );
247         }
248 
249         return status.toString();
250     }

253     private void setProjectProperties( String prefix, Map<String, Object> entryProperties ) {
254         logDebugInfo( " setting properties" );
255         for ( Map.Entry<String, Object> entryProperty : entryProperties.entrySet() ) {
256             setProjectProperty( prefix + '.' + entryProperty.getKey(), String.valueOf( entryProperty.getValue() ) );
257         }
258     }

260     private void setProjectProperty( String name, String value ) {
261         Properties projectProperties = project.getProperties();
262         if ( projectProperties.getProperty( name ) != null ) {
263             logWarning( "the \"" + name + "\" property is already defined, its value will be overwritten. Consider another value for the entry properties prefix." );
264         }
265         projectProperties.setProperty( name, value );
266         logDebugInfo( "  " + name + " = " + value );
267     }

297     private final class EntryStatusHandler implements ISVNStatusHandler {
298 
299         private long maximumRevisionNumber;
300 
301         private long minimumRevisionNumber;
302 
303 
304         private long maximumCommittedRevisionNumber;
305 
306 
307         private final Set<SVNStatusType> localStatusTypes;
308 
309         private final Set<SVNStatusType> remoteStatusTypes;
310 
311 
312         private EntryStatusHandler() {
313             maximumRevisionNumber = Long.MIN_VALUE;
314             minimumRevisionNumber = Long.MAX_VALUE;
315 
316             maximumCommittedRevisionNumber = Long.MIN_VALUE;
317 
318             localStatusTypes = new HashSet<SVNStatusType>();
319             remoteStatusTypes = new HashSet<SVNStatusType>();
320         }
321 
322 
323         public void handleStatus( SVNStatus status ) {
324             long revisionNumber = status.getRevision().getNumber();
325             if ( SVNRevision.isValidRevisionNumber( revisionNumber ) ) {
326                 maximumRevisionNumber = Math.max( maximumRevisionNumber, revisionNumber );
327                 if ( revisionNumber != 0 ) {
328                     minimumRevisionNumber = Math.min( minimumRevisionNumber, revisionNumber );
329                 }
330             }
331             long committedRevisionNumber = status.getCommittedRevision().getNumber();
332             if ( SVNRevision.isValidRevisionNumber( committedRevisionNumber ) ) {
333                 maximumCommittedRevisionNumber = Math.max( maximumCommittedRevisionNumber, committedRevisionNumber );
334             }
335 
336             SVNStatusType contentsStatusType = status.getContentsStatus();
337             localStatusTypes.add( contentsStatusType );
338 
339             SVNStatusType propertiesStatusType = status.getPropertiesStatus();
340             localStatusTypes.add( propertiesStatusType );
341 
342             SVNStatusType remoteContentsStatusType = status.getRemoteContentsStatus();
343             remoteStatusTypes.add( remoteContentsStatusType );
344             SVNStatusType remotePropertiesStatusType = status.getRemotePropertiesStatus();
345             remoteStatusTypes.add( remotePropertiesStatusType );
346 
347             boolean entryOutOfDate = !SVNStatusType.STATUS_NONE.equals( remoteContentsStatusType )
348                     || !SVNStatusType.STATUS_NONE.equals( remotePropertiesStatusType );
349 
350             StringBuilder buffer = new StringBuilder();
351             buffer.append( "  " );
352             buffer.append( contentsStatusType.getCode() ).append( propertiesStatusType.getCode() );
353             buffer.append( entryOutOfDate ? '*' : ' ' );
354             buffer.append( ' ' ).append( String.format( "%6d", revisionNumber ) );
355             buffer.append( ' ' ).append( String.format( "%6d", committedRevisionNumber ) );
356             buffer.append( ' ' ).append( status.getFile() );
357             logDebugInfo( buffer.toString() );
358         }
359 
360 
361         public long getMaximumRevisionNumber() {
362             return maximumRevisionNumber == Long.MIN_VALUE ? -1L : maximumRevisionNumber;
363         }
364 
365         public long getMinimumRevisionNumber() {
366             return minimumRevisionNumber == Long.MAX_VALUE ? -1L : minimumRevisionNumber;
367         }
368 
369         public boolean isMixedRevisions() {
370             return getMaximumRevisionNumber() > 0L && getMinimumRevisionNumber() > 0L
371                     && getMaximumRevisionNumber() != getMinimumRevisionNumber();
372         }
373 
374 
375         public long getMaximumCommittedRevisionNumber() {
376             return maximumCommittedRevisionNumber == Long.MIN_VALUE ? -1L : maximumCommittedRevisionNumber;
377         }
378 
379 
380         public Set<SVNStatusType> getLocalStatusTypes() {
381             return new HashSet<SVNStatusType>( localStatusTypes );
382         }
383 
384         public Set<SVNStatusType> getRemoteStatusTypes() {
385             return new HashSet<SVNStatusType>( remoteStatusTypes );
386         }

388     }

390     private static class EntryStatusSymbols {
391 
392         public static final EntryStatusSymbols DEFAULT = new EntryStatusSymbols();
393 
394         public static final EntryStatusSymbols SPECIAL = new EntryStatusSymbols() {
395             @Override
396             public char getStatusSymbol( SVNStatusType svnStatusType ) {
397                 if ( SVNStatusType.STATUS_UNVERSIONED.equals( svnStatusType ) ) {
398                     return 'u';
399                 } else if ( SVNStatusType.STATUS_MISSING.equals( svnStatusType ) ) {
400                     return 'm';
401                 } else if ( SVNStatusType.STATUS_INCOMPLETE.equals( svnStatusType ) ) {
402                     return 'i';
403                 } else if ( SVNStatusType.STATUS_OBSTRUCTED.equals( svnStatusType ) ) {
404                     return 'o';
405                 } else {
406                     return super.getStatusSymbol( svnStatusType );
407                 }
408             }
409 
410             @Override
411             public char getOutOfDateSymbol() {
412                 return 'd';
413             }
414         };
415 
416 
417         private EntryStatusSymbols() {
418         }
419 
420 
421         public char getStatusSymbol( SVNStatusType svnStatusType ) {
422             return svnStatusType.getCode();
423         }
424 
425         public char getOutOfDateSymbol() {
426             return '*';
427         }
