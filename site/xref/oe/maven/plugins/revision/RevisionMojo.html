82      /**
83       * Specifies the list of entries to inspect. Each entry has a separate configuration consisting of the local path,
84       * report options and the prefix for the output properties.
85       * <p/>
86       * The following example shows the entry configured with the default properties:
87       * <pre>
88       * &lt;entries&gt;
89       *   &lt;entry&gt;
90       *     &lt;path&gt;${project.basedir}&lt;path&gt;
91       *     &lt;prefix&gt;${project.artifactId}&lt;prefix&gt;
92       *     &lt;depth&gt;infinity&lt;depth&gt;
93       *     &lt;reportUnversioned&gt;true&lt;reportUnversioned&gt;
94       *     &lt;reportIgnored&gt;false&lt;reportIgnored&gt;
95       *     &lt;reportOutOfDate&gt;false&lt;reportOutOfDate&gt;
96       *   &lt;entry&gt;
97       * &lt;entries&gt;
98       * </pre>
99       * <p/>
100      * If entries configuration is not specified then the goal will operate on the default entry with the entry path
101      * equal to the project basedir and the properties prefix equal to the project artifactId.
102      *
103      * @parameter
104      */
105     private Entry[] entries;

115     public void execute() throws MojoExecutionException, MojoFailureException {
116         if ( entries == null ) {
117             logDebug( "the entries configuration is missing, using default values" );
118             // defaulting to the path = project.basedir, prefix = project.artifactId, depth = infinity, reporting unversioned
119             entries = new Entry[] {
120                     new Entry( project.getBasedir(), project.getArtifactId() ),
121             };
122         } else if ( entries.length == 0 ) {
123             throw new MojoExecutionException( "the entries configuration list is empty" );
124         }
125 
126         SVNStatusClient statusClient = SVNClientManager.newInstance().getStatusClient();
127 
128         for ( Entry entry : entries ) {
129             if ( entry.getPath() == null ) {
130                 logDebug( "the entry path is not specified, using project.basedir: " + project.getBasedir() );
131                 entry.setPath( project.getBasedir() );
132             }
133             logInfo( "inspecting " + entry.getPath() );
134             if ( entry.getPrefix() == null ) {
135                 logDebug( "the entry prefix is not specified, using project.artifactId: " + project.getArtifactId() );
136                 entry.setPrefix( project.getArtifactId() );
137             }
138             logDebugInfo( "  prefix = " + entry.getPrefix() );
139             logDebugInfo( "  depth = " + entry.getDepth() );
140             logDebugInfo( "  report unversioned = " + entry.reportUnversioned() );
141             logDebugInfo( "  report ignored = " + entry.reportIgnored() );
142             logDebugInfo( "  report out-of-date = " + entry.reportOutOfDate() );
143             entry.validate();
144             Map<String, Object> entryProperties = getEntryProperties( entry, statusClient );
145             setProjectProperties( entry.getPrefix(), entryProperties );
146         }
147     }

150     private Map<String, Object> getEntryProperties( Entry entry, SVNStatusClient statusClient ) throws MojoExecutionException {
151         try {
152             SVNStatus status = statusClient.doStatus( entry.getPath(), false );
153             return createVersionedEntryProperties( entry, status, statusClient );
154         } catch ( SVNException e ) {
155             SVNErrorCode errorCode = e.getErrorMessage() == null ? null : e.getErrorMessage().getErrorCode();
156             if ( SVNErrorCode.WC_NOT_DIRECTORY.equals( errorCode ) ) {
157                 if ( getLog().isDebugEnabled() ) {
158                     getLog().debug( e );
159                 }
160                 return createUnversionedEntryProperties();
161             } else {
162                 if ( getLog().isErrorEnabled() ) {
163                     getLog().error( e );
164                 }
165                 return createSpecialEntryProperties( entry, statusClient );
166             }
167         }
168     }

170     private Map<String, Object> createUnversionedEntryProperties() {
171         Map<String, Object> properties = new LinkedHashMap<String, Object>();
172         properties.put( "repository", "" );
173         properties.put( "path", "" );
174         properties.put( "revision", -1L );
175         properties.put( "mixedRevisions", "false" );
176         properties.put( "committedRevision", -1L );
177         properties.put( "committedDate", "" );
178         properties.put( "status", EntryStatusSymbols.DEFAULT.getStatusSymbol( SVNStatusType.STATUS_UNVERSIONED ) );
179         properties.put( "specialStatus", EntryStatusSymbols.SPECIAL.getStatusSymbol( SVNStatusType.STATUS_UNVERSIONED ) );
180         return properties;
181     }

183     private Map<String, Object> createVersionedEntryProperties( Entry entry, SVNStatus status, SVNStatusClient statusClient ) throws MojoExecutionException {
184         VersionedEntryStatusHandler entryStatusHandler = new VersionedEntryStatusHandler();
185         entryStatusHandler.setRepositoryProperties( status.getEntry() );
186         try {
187             logDebugInfo( " collecting status information" );
188             statusClient.doStatus( entry.getPath(),
189                     SVNRevision.WORKING, SVNDepth.fromString( entry.getDepth() ),
190                     entry.reportOutOfDate(), true, entry.reportIgnored(), false,
191                     entryStatusHandler,
192                     null );
193         } catch ( SVNException e ) {
194             throw new MojoExecutionException( e.getMessage(), e );
195         }
196 
197         Map<String, Object> properties = new LinkedHashMap<String, Object>();
198         properties.put( "repository", entryStatusHandler.getRepositoryRoot() );
199         properties.put( "path", entryStatusHandler.getRepositoryPath() );
200         properties.put( "revision", entryStatusHandler.getMaximumRevisionNumber() );
201         properties.put( "mixedRevisions", entryStatusHandler.isMixedRevisions() );
202         properties.put( "committedRevision", entryStatusHandler.getMaximumCommittedRevisionNumber() );
203         properties.put( "committedDate", entryStatusHandler.getMaximumCommittedRevisionDate() );
204         properties.put( "status", createStatusString( entry, entryStatusHandler.getLocalStatusTypes(), entryStatusHandler.getRemoteStatusTypes(), EntryStatusSymbols.DEFAULT ) );
205         properties.put( "specialStatus", createStatusString( entry, entryStatusHandler.getLocalStatusTypes(), entryStatusHandler.getRemoteStatusTypes(), EntryStatusSymbols.SPECIAL ) );
206         return properties;
207     }

209     private Map<String, Object> createSpecialEntryProperties( final Entry entry, SVNStatusClient statusClient ) throws MojoExecutionException {
210         // hack: obtain the status of a single "problem" entry by asking the entry's parent
211         SpecialEntryStatusHandler entryStatusHandler = new SpecialEntryStatusHandler( entry.getPath().getName() );
212         try {
213             logDebugInfo( " collecting status information from the entry parent" );
214             statusClient.doStatus( entry.getPath().getParentFile(),
215                     SVNRevision.WORKING, SVNDepth.IMMEDIATES,
216                     false, true, entry.reportIgnored(), false,
217                     entryStatusHandler,
218                     null );
219         } catch ( SVNException e ) {
220             throw new MojoExecutionException( e.getMessage(), e );
221         }
222 
223         Map<String, Object> properties = new LinkedHashMap<String, Object>();
224         properties.put( "repository", entryStatusHandler.getRepositoryRoot() );
225         properties.put( "path", entryStatusHandler.getRepositoryPath() );
226         properties.put( "revision", entryStatusHandler.getMaximumRevisionNumber() );
227         properties.put( "mixedRevisions", entryStatusHandler.isMixedRevisions() );
228         properties.put( "committedRevision", entryStatusHandler.getMaximumCommittedRevisionNumber() );
229         properties.put( "committedDate", entryStatusHandler.getMaximumCommittedRevisionDate() );
230         properties.put( "status", createStatusString( entry, entryStatusHandler.getLocalStatusTypes(), entryStatusHandler.getRemoteStatusTypes(), EntryStatusSymbols.DEFAULT ) );
231         properties.put( "specialStatus", createStatusString( entry, entryStatusHandler.getLocalStatusTypes(), entryStatusHandler.getRemoteStatusTypes(), EntryStatusSymbols.SPECIAL ) );
232         return properties;
233     }

236     private void setProjectProperties( String prefix, Map<String, Object> entryProperties ) {
237         logDebugInfo( " setting properties" );
238         for ( Map.Entry<String, Object> entryProperty : entryProperties.entrySet() ) {
239             setProjectProperty( prefix + '.' + entryProperty.getKey(), String.valueOf( entryProperty.getValue() ) );
240         }

243     private void setProjectProperty( String name, String value ) {
244         Properties projectProperties = project.getProperties();
245         if ( projectProperties.getProperty( name ) != null ) {
246             logWarning( "the \"" + name + "\" property is already defined, its value will be overwritten. Consider another value for the entry properties prefix." );
247         }
248         projectProperties.setProperty( name, value );
249         logDebugInfo( "  " + name + " = " + value );
250     }

253     private String createStatusString( Entry entry, Set<SVNStatusType> localStatusTypes, Set<SVNStatusType> remoteStatusTypes, EntryStatusSymbols symbols ) {
254         StringBuilder status = new StringBuilder();
255 
256         localStatusTypes.remove( SVNStatusType.STATUS_NONE );
257         localStatusTypes.remove( SVNStatusType.STATUS_NORMAL );
258         if ( localStatusTypes.remove( SVNStatusType.STATUS_ADDED ) ) {
259             status.append( symbols.getStatusSymbol( SVNStatusType.STATUS_ADDED ) );
260         }
261         if ( localStatusTypes.remove( SVNStatusType.STATUS_CONFLICTED ) ) {
262             status.append( symbols.getStatusSymbol( SVNStatusType.STATUS_CONFLICTED ) );
263         }
264         if ( localStatusTypes.remove( SVNStatusType.STATUS_DELETED ) ) {
265             status.append( symbols.getStatusSymbol( SVNStatusType.STATUS_DELETED ) );
266         }
267         if ( localStatusTypes.remove( SVNStatusType.STATUS_IGNORED ) && entry.reportIgnored() ) {
268             status.append( symbols.getStatusSymbol( SVNStatusType.STATUS_IGNORED ) );
269         }
270         if ( localStatusTypes.remove( SVNStatusType.STATUS_MODIFIED ) ) {
271             status.append( symbols.getStatusSymbol( SVNStatusType.STATUS_MODIFIED ) );
272         }
273         if ( localStatusTypes.remove( SVNStatusType.STATUS_REPLACED ) ) {
274             status.append( symbols.getStatusSymbol( SVNStatusType.STATUS_REPLACED ) );
275         }
276         if ( localStatusTypes.remove( SVNStatusType.STATUS_EXTERNAL ) ) {
277             status.append( symbols.getStatusSymbol( SVNStatusType.STATUS_EXTERNAL ) );
278         }
279         if ( localStatusTypes.remove( SVNStatusType.STATUS_UNVERSIONED ) && entry.reportUnversioned() ) {
280             status.append( symbols.getStatusSymbol( SVNStatusType.STATUS_UNVERSIONED ) );
281         }
282         if ( localStatusTypes.remove( SVNStatusType.STATUS_MISSING ) ) {
283             status.append( symbols.getStatusSymbol( SVNStatusType.STATUS_MISSING ) );
284         }
285         if ( localStatusTypes.remove( SVNStatusType.STATUS_INCOMPLETE ) ) {
286             status.append( symbols.getStatusSymbol( SVNStatusType.STATUS_INCOMPLETE ) );
287         }
288         if ( localStatusTypes.remove( SVNStatusType.STATUS_OBSTRUCTED ) ) {
289             status.append( symbols.getStatusSymbol( SVNStatusType.STATUS_OBSTRUCTED ) );
290         }
291         if ( !localStatusTypes.isEmpty() ) {
292             // future proofing
293             logWarning( "the following svn statuses are not taken into account: " + localStatusTypes );
294         }
295 
296         remoteStatusTypes.remove( SVNStatusType.STATUS_NONE );
297         if ( !remoteStatusTypes.isEmpty() && entry.reportOutOfDate() ) {
298             status.append( symbols.getOutOfDateSymbol() );
299         }
300 
301         return status.toString();
302     }

332     private abstract class AbstractStatusHandler implements ISVNStatusHandler {
333 
334         private String repositoryRoot;
335 
336         private String repositoryPath;
337 
338 
339         private long maximumRevisionNumber;
340 
341         private long minimumRevisionNumber;
342 
343 
344         private long maximumCommittedRevisionNumber;
345 
346         private Date maximumCommittedRevisionDate;
347 
348 
349         private final Set<SVNStatusType> localStatusTypes;
350 
351         private final Set<SVNStatusType> remoteStatusTypes;
352 
353 
354         private AbstractStatusHandler() {
355             repositoryRoot = "";
356             repositoryPath = "";
357 
358             maximumRevisionNumber = Long.MIN_VALUE;
359             minimumRevisionNumber = Long.MAX_VALUE;
360 
361             maximumCommittedRevisionNumber = Long.MIN_VALUE;
362 
363             localStatusTypes = new HashSet<SVNStatusType>();
364             remoteStatusTypes = new HashSet<SVNStatusType>();
365         }
366 
367 
368         protected void setRepositoryProperties( SVNEntry svnEntry ) {
369             repositoryRoot = svnEntry == null ? "" : svnEntry.getRepositoryRoot();
370             repositoryPath = svnEntry == null || svnEntry.getURL() == null ? "" : svnEntry.getURL().substring( repositoryRoot.length() );
371             if ( repositoryPath.startsWith( "/" ) ) {
372                 repositoryPath = repositoryPath.substring( 1 );
373             }
374         }
375 
376         protected void appendStatus( SVNStatus status ) {
377             long revisionNumber = status.getRevision().getNumber();
378             if ( SVNRevision.isValidRevisionNumber( revisionNumber ) ) {
379                 if ( maximumRevisionNumber < revisionNumber ) {
380                     maximumRevisionNumber = revisionNumber;
381                 }
382                 if ( revisionNumber != 0 && minimumRevisionNumber > revisionNumber ) {
383                     minimumRevisionNumber = revisionNumber;
384                 }
385             }
386             long committedRevisionNumber = status.getCommittedRevision().getNumber();
387             if ( SVNRevision.isValidRevisionNumber( committedRevisionNumber ) && maximumCommittedRevisionNumber < committedRevisionNumber ) {
388                 maximumCommittedRevisionNumber = committedRevisionNumber;
389                 maximumCommittedRevisionDate = status.getCommittedDate();
390             }
391 
392             SVNStatusType contentsStatusType = status.getContentsStatus();
393             localStatusTypes.add( contentsStatusType );
394 
395             SVNStatusType propertiesStatusType = status.getPropertiesStatus();
396             localStatusTypes.add( propertiesStatusType );
397 
398             SVNStatusType remoteContentsStatusType = status.getRemoteContentsStatus();
399             remoteStatusTypes.add( remoteContentsStatusType );
400             SVNStatusType remotePropertiesStatusType = status.getRemotePropertiesStatus();
401             remoteStatusTypes.add( remotePropertiesStatusType );
402 
403             boolean entryOutOfDate = !SVNStatusType.STATUS_NONE.equals( remoteContentsStatusType )
404                     || !SVNStatusType.STATUS_NONE.equals( remotePropertiesStatusType );
405 
406             StringBuilder buffer = new StringBuilder();
407             buffer.append( "  " );
408             buffer.append( contentsStatusType.getCode() ).append( propertiesStatusType.getCode() );
409             buffer.append( entryOutOfDate ? '*' : ' ' );
410             buffer.append( ' ' ).append( String.format( "%6d", revisionNumber ) );
411             buffer.append( ' ' ).append( String.format( "%6d", committedRevisionNumber ) );
412             buffer.append( ' ' ).append( status.getFile() );
413             logDebugInfo( buffer.toString() );
414         }
415 
416 
417         public String getRepositoryRoot() {
418             return repositoryRoot;
419         }
420 
421         public String getRepositoryPath() {
422             return repositoryPath;
423         }
424 
425 
426         public long getMaximumRevisionNumber() {
427             return maximumRevisionNumber == Long.MIN_VALUE ? -1L : maximumRevisionNumber;
428         }
429 
430         public long getMinimumRevisionNumber() {
431             return minimumRevisionNumber == Long.MAX_VALUE ? -1L : minimumRevisionNumber;
432         }
433 
434         public boolean isMixedRevisions() {
435             return getMaximumRevisionNumber() > 0L && getMinimumRevisionNumber() > 0L
436                     && getMaximumRevisionNumber() != getMinimumRevisionNumber();
437         }
438 
439 
440         public long getMaximumCommittedRevisionNumber() {
441             return maximumCommittedRevisionNumber == Long.MIN_VALUE ? -1L : maximumCommittedRevisionNumber;
442         }
443 
444         public String getMaximumCommittedRevisionDate() {
445             return maximumCommittedRevisionDate == null ?
446                     "" :
447                     String.format( Locale.ENGLISH, "%tF %<tT %<tz (%<ta, %<td %<tb %<tY)", maximumCommittedRevisionDate );
448         }
449 
450 
451         public Set<SVNStatusType> getLocalStatusTypes() {
452             return new HashSet<SVNStatusType>( localStatusTypes );
453         }
454 
455         public Set<SVNStatusType> getRemoteStatusTypes() {
456             return new HashSet<SVNStatusType>( remoteStatusTypes );
457         }

472     private final class SpecialEntryStatusHandler extends AbstractStatusHandler {
473 
474         private final String entryName;
475 
476         private SpecialEntryStatusHandler( String entryName ) {
477             if ( entryName == null ) {
478                 throw new IllegalArgumentException( "entryName is null" );
479             }
480             this.entryName = entryName;
481         }
482 
483         public void handleStatus( SVNStatus status ) {
484             SVNEntry svnEntry = status.getEntry();
485             if ( svnEntry != null && entryName.equals( svnEntry.getName() ) ) {
486                 setRepositoryProperties( svnEntry );
487                 appendStatus( status );
488             }
489         }
