85       * Specifies the list of entries to inspect. Each entry has a separate configuration consisting of the local path,
86       * report options and the prefix for the output properties.
87       * <p/>
88       * The following example shows the entry configured with the default properties:
89       * <pre>
90       * &lt;entries&gt;
91       *   &lt;entry&gt;
92       *     &lt;path&gt;${project.basedir}&lt;/path&gt;
93       *     &lt;prefix&gt;${project.artifactId}&lt;/prefix&gt;
94       *     &lt;depth&gt;infinity&lt;/depth&gt;
95       *     &lt;reportUnversioned&gt;true&lt;/reportUnversioned&gt;
96       *     &lt;reportIgnored&gt;false&lt;/reportIgnored&gt;
97       *     &lt;reportOutOfDate&gt;false&lt;/reportOutOfDate&gt;
98       *   &lt;/entry&gt;
99       * &lt;/entries&gt;
100      * </pre>
101      * <p/>
102      * If the entries configuration is not specified then the goal will operate on the default entry with the entry path
103      * equal to the project basedir and the properties prefix equal to the project artifactId.
104      *
105      * @parameter
106      */

125     public void execute() throws MojoExecutionException, MojoFailureException {
126         if ( entries == null || entries.length == 0 ) {
127             if ( getLog().isDebugEnabled() ) {
128                 getLog().debug( "configuration/entries is not specified or empty, using default entry" );
129             }
130             entries = new Entry[] {
131                     new Entry( project.getBasedir(), project.getArtifactId() ),
132             };
133         }
134 
135         for ( Entry entry : entries ) {
136             if ( entry.getPath() == null ) {
137                 entry.setPath( project.getBasedir() );
138             }
139             if ( entry.getPrefix() == null ) {
140                 entry.setPrefix( project.getArtifactId() );
141             }
142             processEntry( entry );
143         }
144     }

147     private void processEntry( Entry entry ) throws MojoExecutionException {
148         if ( getLog().isInfoEnabled() ) {
149             getLog().info( format( "inspecting %s %s", entry.getPath().isFile() ? "file" : entry.getPath().isDirectory() ? "directory" : "path", entry.getPath() ) );
150         }
151         logDebugInfo( format( "  prefix = %s", entry.getPrefix() ) );
152         logDebugInfo( format( "  depth = %s", entry.getDepth() ) );
153         logDebugInfo( format( "  report unversioned = %s", entry.reportUnversioned() ) );
154         logDebugInfo( format( "  report ignored = %s", entry.reportIgnored() ) );
155         logDebugInfo( format( "  report out-of-date = %s", entry.reportOutOfDate() ) );
156 
157         logDebugInfo( "calculating properties" );
158         SvnOperationFactory operationFactory = new SvnOperationFactory();
159         StatusHandler statusHandler = new StatusHandler( entry );
160         try {
161             logDebugInfo( format( "  wc format = %s", SvnOperationFactory.detectWcGeneration( entry.getPath(), true ) ) );
162             fillStatus( entry, operationFactory, statusHandler );
163             fillInfo( entry, operationFactory, statusHandler );
164         } catch ( SVNException e ) {
165             if ( e.getErrorMessage() != null && ( SVNErrorCode.WC_NOT_WORKING_COPY.equals( e.getErrorMessage().getErrorCode() ) || SVNErrorCode.WC_PATH_NOT_FOUND.equals( e.getErrorMessage().getErrorCode() ) ) ) {
166                 statusHandler.resetProperties( true );
167             } else if ( failOnError ) {
168                 throw new MojoExecutionException( e.getMessage(), e );
169             } else {
170                 if ( getLog().isErrorEnabled() ) {
171                     getLog().error( e );
172                 }
173                 statusHandler.resetProperties();
174             }
175         }
176         setProjectProperties( entry.getPrefix(), statusHandler.createProperties() );
177     }

180     private void fillStatus( Entry entry, SvnOperationFactory operationFactory, StatusHandler statusHandler ) throws SVNException {
181         SvnGetStatus statusOperation = operationFactory.createGetStatus();
182         statusOperation.setSingleTarget( SvnTarget.fromFile( entry.getPath() ) );
183         statusOperation.setDepth( SVNDepth.fromString( entry.getDepth() ) );
184         statusOperation.setRevision( SVNRevision.WORKING );
185         statusOperation.setReportAll( true );
186         statusOperation.setReportIgnored( entry.reportIgnored() );
187         statusOperation.setRemote( entry.reportOutOfDate() );
188         statusOperation.setReceiver( statusHandler );
189         statusOperation.run();
190     }
191 
192     private void fillInfo( Entry entry, SvnOperationFactory operationFactory, StatusHandler statusHandler ) throws SVNException {
193         if ( statusHandler.repositoryPath == null || statusHandler.repositoryPath.length() == 0 ) {
194             SvnGetInfo infoOperation = operationFactory.createGetInfo();
195             infoOperation.setSingleTarget( SvnTarget.fromFile( entry.getPath() ) );
196             SvnInfo infoResult = infoOperation.run();
197             statusHandler.repositoryRoot = infoResult.getRepositoryRootUrl().toString();
198             statusHandler.repositoryPath = infoResult.getUrl().toString().substring( statusHandler.repositoryRoot.length() + 1 );
199         }
200     }

203     private void setProjectProperties( String prefix, Map<String, Object> properties ) {
204         logDebugInfo( "setting properties" );
205         for ( Map.Entry<String, Object> entryProperty : properties.entrySet() ) {
206             setProjectProperty( prefix + '.' + entryProperty.getKey(), String.valueOf( entryProperty.getValue() ) );
207         }
208     }

210     private void setProjectProperty( String name, String value ) {
211         Properties projectProperties = project.getProperties();
212         if ( projectProperties.getProperty( name ) != null ) {
213             logDebugWarning( format( "the \"%s\" property is already defined and will be overwritten. The possible causes for this are:%n" +
214                     "  - the plugin configuration contains two or more entries with the same prefix.%n" +
215                     "  - the plugin runs multiple times with the same configuration.%n" +
216                     "  - the property is already defined in the POM or by some other plugin.",
217                     name ) );
218         }
219         projectProperties.setProperty( name, value );
220         logDebugInfo( format( "  %s = %s", name, value ) );
221     }

249     private final class StatusHandler implements ISvnObjectReceiver<SvnStatus> {
250 
251         private final Entry entry;
252 
253         private String repositoryRoot;
254 
255         private String repositoryPath;
256 
257         private long maximumRevision;
258 
259         private long minimumRevision;
260 
261         private long committedRevision;
262 
263         private Date committedDate;
264 
265         private Set<SVNStatusType> localStatusTypes;
266 
267         private boolean outOfDate;
268 
269 
270         private StatusHandler( Entry entry ) {
271             if ( entry == null ) {
272                 throw new IllegalArgumentException( "{entry} is null" );
273             }
274             this.entry = entry;
275             resetProperties();
276         }
277 
278 
279         public void receive( SvnTarget target, SvnStatus status ) throws SVNException {
280             logDebugInfo( format( "  %s%s%s %s%s%s  %6s %6s %6s  %s (%s %s)",
281                     status.getNodeStatus().getCode(), status.getPropertiesStatus().getCode(), status.getTextStatus().getCode(),
282                     status.getRepositoryNodeStatus().getCode(), status.getRepositoryPropertiesStatus().getCode(), status.getRepositoryTextStatus().getCode(),
283                     status.getRevision(), status.getChangedRevision(), status.getRepositoryChangedRevision(),
284                     target.getPathOrUrlString(),
285                     status.getRepositoryRootUrl(), status.getRepositoryRelativePath()
286             ) );
287 
288             if ( repositoryRoot == null ) {
289                 repositoryRoot = status.getRepositoryRootUrl() == null ? "" : status.getRepositoryRootUrl().toString();
290                 repositoryPath = status.getRepositoryRelativePath();
291             }
292 
293             long targetRevision = status.getRevision();
294             if ( targetRevision >= 0L ) {
295                 if ( maximumRevision < targetRevision ) {
296                     maximumRevision = targetRevision;
297                 }
298                 if ( targetRevision > 0L && minimumRevision > targetRevision ) {
299                     minimumRevision = targetRevision;
300                 }
301             }
302             long targetCommittedRevision = status.getChangedRevision();
303             if ( targetCommittedRevision >= 0L && committedRevision < targetCommittedRevision ) {
304                 committedRevision = targetCommittedRevision;
305                 committedDate = status.getChangedDate();
306             }
307 
308             localStatusTypes.add( status.getNodeStatus() );
309             if ( SVNStatusType.STATUS_NORMAL.equals( status.getNodeStatus() ) ) {
310                 localStatusTypes.add( status.getPropertiesStatus() );
311             }
312 
313             if ( status.getRepositoryChangedRevision() > targetCommittedRevision ) {
314                 outOfDate = true;
315             }
316         }
317 
318         public void resetProperties() {
319             resetProperties( false );
320         }
321 
322         public void resetProperties( boolean forceUnversioned ) {
323             repositoryRoot = null;
324             repositoryPath = null;
325 
326             maximumRevision = Long.MIN_VALUE;
327             minimumRevision = Long.MAX_VALUE;
328 
329             committedRevision = Long.MIN_VALUE;
330             committedDate = null;
331 
332             localStatusTypes = new HashSet<SVNStatusType>();
333             outOfDate = false;
334 
335             if ( forceUnversioned ) {
336                 localStatusTypes.add( SVNStatusType.STATUS_UNVERSIONED );
337             }
338         }
339 
340         public Map<String, Object> createProperties() {
341             Map<String, Object> properties = new LinkedHashMap<String, Object>();
342             properties.put( "repository", repositoryRoot == null ? "" : repositoryRoot );
343             properties.put( "path", repositoryPath == null ? "" : repositoryPath );
344             properties.put( "revision", maximumRevision == Long.MIN_VALUE ? -1L : maximumRevision );
345             properties.put( "mixedRevisions", maximumRevision > 0L && minimumRevision > 0L && maximumRevision != minimumRevision );
346             properties.put( "committedRevision", committedRevision == Long.MIN_VALUE ? -1L : committedRevision );
347             properties.put( "committedDate", committedDate == null ? "" : format( Locale.ENGLISH, "%tF %<tT %<tz (%<ta, %<td %<tb %<tY)", committedDate ) );
348             properties.put( "status", createStatusString( EntryStatusSymbols.DEFAULT ) );
349             properties.put( "specialStatus", createStatusString( EntryStatusSymbols.SPECIAL ) );
350             return properties;
351         }
352 
353         private String createStatusString( EntryStatusSymbols symbols ) {
354             StringBuilder status = new StringBuilder();
355 
356             Set<SVNStatusType> statusTypes = new HashSet<SVNStatusType>( localStatusTypes );
357             statusTypes.remove( SVNStatusType.STATUS_NONE );
358             statusTypes.remove( SVNStatusType.STATUS_NORMAL );
359             if ( statusTypes.remove( SVNStatusType.STATUS_ADDED ) ) {
360                 status.append( symbols.getStatusSymbol( SVNStatusType.STATUS_ADDED ) );
361             }
362             if ( statusTypes.remove( SVNStatusType.STATUS_CONFLICTED ) ) {
363                 status.append( symbols.getStatusSymbol( SVNStatusType.STATUS_CONFLICTED ) );
364             }
365             if ( statusTypes.remove( SVNStatusType.STATUS_DELETED ) ) {
366                 status.append( symbols.getStatusSymbol( SVNStatusType.STATUS_DELETED ) );
367             }
368             if ( statusTypes.remove( SVNStatusType.STATUS_IGNORED ) && entry.reportIgnored() ) {
369                 status.append( symbols.getStatusSymbol( SVNStatusType.STATUS_IGNORED ) );
370             }
371             if ( statusTypes.remove( SVNStatusType.STATUS_MODIFIED ) ) {
372                 status.append( symbols.getStatusSymbol( SVNStatusType.STATUS_MODIFIED ) );
373             }
374             if ( statusTypes.remove( SVNStatusType.STATUS_REPLACED ) ) {
375                 status.append( symbols.getStatusSymbol( SVNStatusType.STATUS_REPLACED ) );
376             }
377             if ( statusTypes.remove( SVNStatusType.STATUS_EXTERNAL ) ) {
378                 status.append( symbols.getStatusSymbol( SVNStatusType.STATUS_EXTERNAL ) );
379             }
380             if ( statusTypes.remove( SVNStatusType.STATUS_UNVERSIONED ) && entry.reportUnversioned() ) {
381                 status.append( symbols.getStatusSymbol( SVNStatusType.STATUS_UNVERSIONED ) );
382             }
383             if ( statusTypes.remove( SVNStatusType.STATUS_MISSING ) ) {
384                 status.append( symbols.getStatusSymbol( SVNStatusType.STATUS_MISSING ) );
385             }
386             if ( statusTypes.remove( SVNStatusType.STATUS_INCOMPLETE ) ) {
387                 status.append( symbols.getStatusSymbol( SVNStatusType.STATUS_INCOMPLETE ) );
388             }
389             if ( statusTypes.remove( SVNStatusType.STATUS_OBSTRUCTED ) ) {
390                 status.append( symbols.getStatusSymbol( SVNStatusType.STATUS_OBSTRUCTED ) );
391             }
392             if ( !statusTypes.isEmpty() ) {
393                 if ( getLog().isWarnEnabled() ) {
394                     getLog().warn( format( "the following svn statuses are not taken into account: %s", statusTypes ) );
395                 }
396             }
397 
398             if ( outOfDate && entry.reportOutOfDate() ) {
399                 status.append( symbols.getOutOfDateSymbol() );
400             }
401 
402             return status.toString();
403         }
